1   /**
2    * Luokka, joka määrittelee korttipakan, josta kortteja aletaan nostaa pelissä.
3    * Pelipaksta voi vain nostaa kortin.
4    */
5   package napoleoninhauta.pasianssi.pinot;
6   
7   import java.util.ArrayList;
8   import napoleoninhauta.pasianssi.pakka.Kortti;
9   
10  public class Pelipakka {
11  
12      private ArrayList<Kortti> kortit;
13  
14      public Pelipakka(ArrayList<Kortti> kortit) {
15          this.kortit = kortit;
16      }
17  
18      /**
19       * Metodi palauttaa pakassa olevien korttien lukumäärän.
20       *
21       * @return pakan koko
22       */
23      public int palautaMaara() {
24          return this.kortit.size();
25      }
26  
27      /**
28       * Metodi palauttaa pakan päälimmäisen kortin ja poistaa sen pakasta. Jos
29       * pakassa ei ole kortteja, metodi vain palauttaa null.
30       *
31       * @return päällimäinen kortti tai null
32       */
33      public Kortti nostaKortti() {
34          if (!kortit.isEmpty()) {
35              Kortti eka = kortit.get(0);
36              kortit.remove(0);
37              return eka;
38          }
39          return null;
40      }
41  
42  }
