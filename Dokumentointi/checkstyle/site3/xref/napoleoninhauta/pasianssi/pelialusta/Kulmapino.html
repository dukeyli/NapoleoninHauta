1   /**
2    * Pinon perivä luokka, joka määrittelee pinon, johon voidaan laittaa kortteja joiden
3    * arvo on välillä 7-13. Pinoon voi laittaa vain yhden arvoista korttia kerrallaan, ensin
4    * seiskaa, sitten kasia jne.
5    */
6   package napoleoninhauta.pasianssi.pelialusta;
7   
8   import napoleoninhauta.pasianssi.pakka.Kortti;
9   
10  public class Kulmapino extends Pino {
11  
12      private int arvo;
13  
14      public Kulmapino() {
15          this.arvo = 7;
16      }
17  
18      /**
19       * Metodi palauttaa attribuutin arvo.
20       *
21       * @return kokonaislukuarvoinen attribuutti
22       */
23      public int palautaArvo() {
24          return this.arvo;
25      }
26  
27      /**
28       * Metodi laittaa parametrina annetun kortin pinoon päälimmäiseksi ja
29       * kasvattaa arvo-attribuuttia yhdellä, jos kortin arvo oikea ja pinossa ei
30       * ole jo kyseistä korttia. Kortin arvo on oikea, jos se on sama kuin
31       * arvo-attribuutin arvo.
32       *
33       * @param kortti pinosta, pakasta tai jemmasta otettava kortti.
34       *
35       */
36      @Override
37      public void asetaKortti(Kortti kortti) {
38          if (kortti.palautaArvo() == arvo && !super.palautaKortit().contains(kortti)) {
39              super.palautaKortit().add(0, kortti);
40              arvo++;
41          }
42      }
43  }
