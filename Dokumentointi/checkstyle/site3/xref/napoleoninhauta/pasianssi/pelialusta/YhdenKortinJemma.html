1   /**
2    * Luokka määrittelee pinon, johon voi liittyä tasan yksi kortti kerrallaan.
3    * Paikkaan voi laittaa kortin ja paikasta voi ottaa kortin pois, jolloin
4    * paikka vapautuu.
5    */
6   package napoleoninhauta.pasianssi.pelialusta;
7   
8   import napoleoninhauta.pasianssi.pakka.Kortti;
9   
10  public class YhdenKortinJemma {
11  
12      private Kortti paikka;
13  
14      public YhdenKortinJemma() {
15      }
16  
17      /**
18       * Metodi laittaa parametrina annetun kortin jemmaan, jos jemmassa ei ole
19       * ennestään korttia
20       *
21       * @param kortti pinosta, pakasta tai jemmasta otettava kortti.
22       *
23       */
24      public void asetaKortti(Kortti kortti) {
25          if (paikka == null) {
26              this.paikka = kortti;
27          }
28      }
29  
30      /**
31       * Jos jemma ei ole tyhja, metodi palauttaa jemman kortin ja poistaa sen
32       * jemmasta, eli laittaa jemmaan arvon null. Muulloin metodi vain palauttaa
33       * null.
34       *
35       * @return jemman kortti tai null
36       */
37      public Kortti otaKortti() {
38          if (paikka != null) {
39              Kortti kortti = paikka;
40              paikka = null;
41              return kortti;
42          }
43          return null;
44      }
45  
46      /**
47       * Metodi palauttaa jemman kortin String-esityksen, tai "tyhja", jos jemma
48       * on tyhjä.
49       *
50       * @return jemman kortin String-esitys tai "tyhja"
51       */
52      public String palautaKortti() {
53          if (paikka == null) {
54              return "tyhja";
55          }
56          return this.paikka.toString();
57      }
58  
59  }
