1   /**
2    * Luokka, joka määrittelee korttipakan, josta kortteja aletaan nostaa pelissä.
3    * Pelipaksta voi vain nostaa kortin.
4    */
5   package napoleoninhauta.logiikka.pinot;
6   
7   import java.util.ArrayList;
8   import napoleoninhauta.pakka.Kortti;
9   import napoleoninhauta.pakka.PakanLuoja;
10  
11  public class Pelipakka {
12  
13      private ArrayList<Kortti> kortit;
14  
15      /**
16       * Konstruktori määrittää attribuutin pelipakaksi annetun parametrin avulla.
17       *
18       * @param kortit Pakanluojan tekemä pakka
19       */
20      public Pelipakka(ArrayList<Kortti> kortit) {
21          this.kortit = kortit;
22      }
23  
24      /**
25       * Metodi palauttaa pakassa olevien korttien lukumäärän.
26       *
27       * @return pakan koko
28       */
29      public int getMaara() {
30          return this.kortit.size();
31      }
32  
33      /**
34       * Metodi palauttaa pakan päälimmäisen kortin ja poistaa sen pakasta. Jos
35       * pakassa ei ole kortteja, metodi vain palauttaa null.
36       *
37       * @return päällimäinen kortti tai null
38       */
39      public Kortti nostaKortti() {
40          if (!kortit.isEmpty()) {
41              Kortti eka = kortit.get(0);
42              kortit.remove(0);
43              return eka;
44          }
45          return null;
46      }
47  
48      /**
49       * Metodi laittaa parametrina annetun kortin pinoon päällimmäiseksi. Metodi
50       * on noston olemassa noston perumista varten.
51       *
52       * @param kortti kortti pelipakasta.
53       *
54       */
55      public void asetaKortti(Kortti kortti) {
56          kortit.add(0, kortti);
57      }
58  
59      /**
60       * Metodi alustaa pakan uudestaan Pakanluojan avulla.
61       */
62      public void alustaUusiPakka() {
63          PakanLuoja luoja = new PakanLuoja();
64          this.kortit = luoja.alustaPakka();
65      }
66  
67  }
