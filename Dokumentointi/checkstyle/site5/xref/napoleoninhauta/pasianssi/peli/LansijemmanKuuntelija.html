4   package napoleoninhauta.pasianssi.peli;
5   
6   import napoleoninhauta.pasianssi.pelialusta.Paivittaja;
7   import java.awt.event.ActionEvent;
8   import java.awt.event.ActionListener;
9   import napoleoninhauta.pasianssi.pelialusta.Pelialusta;
10  
11  public class LansijemmanKuuntelija implements ActionListener {
12  
13      private Pelialusta alusta;
14      private Paivittaja paivittaja;
15  
16      public LansijemmanKuuntelija(Pelialusta pelialusta, Paivittaja paivittaja) {
17          this.alusta = pelialusta;
18          this.paivittaja = paivittaja;
19      }
20  
21      @Override
22      public void actionPerformed(ActionEvent e) {
23          if (!alusta.getJemmat().getLansi().palautaKortti().equals("tyhja")) {
24              suorita();
25              paivittaja.paivita();
26          }
27      }
28  
29      public void suorita() {
30          int kortinArvo = alusta.getJemmat().getLansi().palautaKortinArvo();
31          if (kortinArvo == alusta.getMuut().getKeskipino().palautaArvo()) {
32              alusta.getMuut().getKeskipino().asetaKortti(alusta.getJemmat().getLansi().otaKortti());
33          }
34          if (alusta.getKulmapinot().palautaArvot().contains(kortinArvo)) {
35              alusta.getKulmapinot().laitaKorttiSopivaanPinoon(alusta.getJemmat().getLansi().otaKortti());
36          }
37      }
38  }
